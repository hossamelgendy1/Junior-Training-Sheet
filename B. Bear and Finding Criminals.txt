//http://codeforces.com/contest/680/problem/B

#include <iostream>
using namespace std;

int main()
{
    int n, index, count=0;
    cin >> n >> index;
    int* arr = new int[n];
    int dis = n-index < index - 1 ? n-index : index-1;
    for (int i = 0; i < n; i++)
    {
        cin >> arr[i];
    }
    if (arr[index - 1] == 1)
        count++;
    int i = 1;
    while (i <= dis && (index + i <= n && index - i >= 0))
    {
        if (arr[index - 1 + i] == 1 && arr[index - 1 - i] == 1)
            count += 2;
        i++;
    }
    while (index + i <= n)
    {
        if (arr[index - 1 + i] == 1)
            count++;
        i++;
    }
    while (index - i >= 0)
    {
        if (arr[index - 1 - i] == 1)
            count++;
        i++;
    }
    cout << count;
}
